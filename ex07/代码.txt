#include <WiFi.h>
#include <WebServer.h>

const char* ssid = "helan";
const char* password = "12345678";

const int LED_PIN = 2;
int brightness = 0;

WebServer server(80);

String makePage() {
  String html = R"rawliteral(
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<title>LED亮度控制</title>
</head>
<body style="text-align:center;margin-top:50px;">
<h1>LED亮度控制</h1>
<p>亮度：<span id="v">0</span></p>
<input type="range" min="0" max="255" oninput="set(this.value)" style="width:80%;height:25px;">
<script>
function set(val){
  document.getElementById("v").innerHTML=val;
  fetch("/set?val="+val);
}
</script>
</body>
</html>
)rawliteral";
  return html;
}

void handleRoot() {
  server.send(200, "text/html; charset=utf-8", makePage());
}

void handleSet() {
  if (server.hasArg("val")) {
    brightness = server.arg("val").toInt();
    brightness = constrain(brightness, 0, 255);
  }
  server.send(200, "text/plain", "OK");
}

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
  analogWrite(LED_PIN, 0);

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(200);
  WiFi.begin(ssid, password);

  Serial.print("正在连接 helan...");
  int timeout = 0;
  while (WiFi.status() != WL_CONNECTED && timeout < 20) {
    delay(500);
    Serial.print(".");
    timeout++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n连接成功");
    Serial.print("访问IP：http://");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\n连接失败，确认2.4G网络与密码");
  }

  server.on("/", handleRoot);
  server.on("/set", handleSet);
  server.begin();
}

void loop() {
  server.handleClient();
  analogWrite(LED_PIN, brightness);
}
