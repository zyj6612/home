// 定义所有LED引脚：2、4、5、21
const int ledPins[] = {2, 4, 5, 21}; 
const int ledCount = sizeof(ledPins) / sizeof(ledPins[0]);

// 时间参数（毫秒）
const int shortOn  = 200;   // 短闪亮时间
const int longOn   = 600;   // 长闪亮时间
const int offTime  = 200;   // 每次闪烁后的熄灭时间
const int letterGap= 500;   // 字母间的间隔
const int sosGap   = 2000;  // 一次SOS结束后的长停顿

// 状态机变量
unsigned long previousMillis = 0;
int currentStep = 0;
bool ledState = false;

// SOS序列：每个状态的动作类型和持续时间
// 0-2: S短闪, 3: 字母间隔, 4-6: O长闪, 7: 字母间隔, 8-10: S短闪, 11: SOS长停顿
const int sosSequence[] = {
  shortOn, offTime, shortOn, offTime, shortOn, offTime,
  letterGap,
  longOn, offTime, longOn, offTime, longOn, offTime,
  letterGap,
  shortOn, offTime, shortOn, offTime, shortOn, offTime,
  sosGap
};
const int sequenceLength = sizeof(sosSequence) / sizeof(sosSequence[0]);

void setup() {
  Serial.begin(115200);
  
  // 初始化所有LED为输出模式
  for(int i=0; i<ledCount; i++){
    pinMode(ledPins[i], OUTPUT);
  }
  
  // 初始状态：熄灭所有LED
  allLedOff();
}

void loop() {
  unsigned long currentMillis = millis();
  
  // 检查当前步骤的时间是否到了
  if (currentMillis - previousMillis >= sosSequence[currentStep]) {
    previousMillis = currentMillis;
    currentStep++;
    
    // 如果序列走完，回到第一步重新开始
    if (currentStep >= sequenceLength) {
      currentStep = 0;
      Serial.println("=== SOS 完成，等待下一轮 ===");
    }
    
    // 控制LED状态：奇数步为熄灭，偶数步为点亮
    if (currentStep % 2 == 0) {
      allLedOn();
      if (currentStep == 0 || currentStep == 6 || currentStep == 14) {
        Serial.println("S - 短闪");
      } else if (currentStep == 8) {
        Serial.println("O - 长闪");
      }
    } else {
      allLedOff();
    }
  }
}

// 函数：所有LED点亮
void allLedOn() {
  for(int i=0; i<ledCount; i++){
    digitalWrite(ledPins[i], HIGH);
  }
}

// 函数：所有LED熄灭
void allLedOff() {
  for(int i=0; i<ledCount; i++){
    digitalWrite(ledPins[i], LOW);
  }
}
